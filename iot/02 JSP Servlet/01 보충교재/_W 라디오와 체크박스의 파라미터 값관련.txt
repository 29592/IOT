
value 가 없는 라디오와 체크 박스는 선택하면 "on"
                                              선택안하면 null
<input type="radio"       name="route" /> 
<input type="checkbox" name="sports" /> 



value 가 있는 라디오와 체크 박스는 선택하면 value 값
                                              선택안하면 null
<input type="radio"      name="route"   value="ad"/> 
<input type="checkbox" name="sports" value="sc"/>
